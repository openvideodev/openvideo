Reference for the main TypeScript types used throughout OpenVideo Core.

## Core Classes

### Core

The main engine class:

```ts
class Core extends EventEmitter {
  store: StoreApi<ProjectStore>; // Zustand store
  playback: PlaybackController; // Playback controls

  constructor(initialState?: Partial<IProject>);

  // Commands
  execute<T>(command: Command<T>): void;
  batch(commands: Command[]): void;
  undo(): void;
  redo(): void;

  // State management
  applyPatch(patches: Patch[]): void;
  reset(project: IProject): void;

  // Convenience methods
  clip: {
    prepare(
      payload: AddClipPayload,
      options?: { objectFit?: "contain" | "cover" },
    ): Promise<AnyClip>;
  };
}
```

## Project Types

### IProject

The root project structure:

```ts
interface IProject {
  settings: IProjectSettings;
  tracks: ITrack[];
  clips: Record<string, AnyClip>;
}

interface IProjectSettings {
  width: number; // Canvas width (default: 1920)
  height: number; // Canvas height (default: 1080)
  fps: number; // Frames per second (default: 30)
  duration: number; // Total duration in microseconds
}
```

### ProjectStore

The full Zustand store state:

```ts
interface ProjectStore extends ProjectState, ProjectActions {}

interface ProjectState extends IProject {
  selectedIds: string[]; // Currently selected clip IDs
  currentTime: number; // Playhead position (µs)
  isPlaying: boolean; // Playback status
  volume: number; // 0-1
  muted: boolean;
  speed: number; // Playback rate (1 = normal)
  scale: IScaleState; // Timeline zoom/pan
  history: HistoryEntry[]; // Undo history
  future: HistoryEntry[]; // Redo stack
}

interface IScaleState {
  zoom: number; // Pixels per microsecond
  pan: number; // Horizontal scroll offset (px)
}
```

```ts
interface ITrack {
```

## Clip Types

### IBaseClip

All clips extend this interface:

```ts
interface IBaseClip {
  id: string;
  type: ClipType;
  name: string;
  timing: IClipTiming;
  transform: IClipTransform;
  src?: string;
  text?: string;
  style?: IBaseClipStyle;
  locked?: boolean;
  metadata?: Record<string, any>;
}

type ClipType = "Video" | "Audio" | "Image" | "Text" | "Transition" | "Caption" | "Effect";
```

### IClipTiming

```ts
interface IClipTiming {
  display: IDisplay; // When on timeline
  trim: ITrim; // Which part of source
  duration: number; // Total duration (µs)
  playbackRate: number; // Speed multiplier
}

interface IDisplay {
  from: number; // Start time on timeline (µs)
  to: number; // End time on timeline (µs)
}

interface ITrim {
  from: number; // Start time in source (µs)
  to: number; // End time in source (µs)
}
```

### IClipTransform

```ts
interface IClipTransform {
  x: number; // Position X (px)
  y: number; // Position Y (px)
  width: number; // Width (px)
  height: number; // Height (px)
  angle: number; // Rotation (degrees)
  zIndex: number; // Layer order
  opacity: number; // 0-1
  flip?: IFlip | null;
}

interface IFlip {
  x: boolean;
  y: boolean;
}
```

### Specific Clip Types

```ts
interface IVideoClip extends IBaseClip {
  type: "Video";
  src: string;
}

interface IAudioClip extends IBaseClip {
  type: "Audio";
  src: string;
}

interface IImageClip extends IBaseClip {
  type: "Image";
  src: string;
}

interface ITextClip extends IBaseClip {
  type: "Text";
  text: string;
  style?: ITextStyle;
}

interface ICaptionClip extends IBaseClip {
  type: "Caption";
  text: string;
  words?: ICaptionWord[];
  style?: ICaptionStyle;
}

type AnyClip =
  | IVideoClip
  | IAudioClip
  | IImageClip
  | ITextClip
  | ICaptionClip
  | ITransitionClip
  | IEffectClip;
```

## Command Types

### Command

```ts
interface Command<T = any> {
  id: string;
  type: string;
  payload: T;
  meta?: CommandMeta;
}

interface CommandMeta {
  userId?: string;
  timestamp?: number;
  source?: "user" | "agent" | "system";
}
```

### Patch

```ts
type PatchOp = "add" | "update" | "remove";

interface Patch {
  op: PatchOp;
  path: string; // JSON Pointer path
  value?: any; // New value
  oldValue?: any; // Previous value
}

type CommandHandler<T = any> = (state: ProjectStore, command: Command<T>) => Patch[];

interface HistoryEntry {
  command: Command;
  patches: Patch[];
  inversePatches: Patch[];
}
```

### DeepPartial

```ts
type DeepPartial<T> = {
```

